CREATE TABLE public.mdl_student_reports
(
    id bigint NOT NULL DEFAULT nextval('mdl_student_reports_id_seq'::regclass),
    timecreated bigint NOT NULL DEFAULT 0,
    timemodified bigint NOT NULL DEFAULT 0,
    userid bigint NOT NULL DEFAULT 0,
    status smallint NOT NULL DEFAULT 0,
    type character varying(10) COLLATE pg_catalog."default" NOT NULL DEFAULT ''::character varying,
    params text COLLATE pg_catalog."default",
    CONSTRAINT mdl_studrepo_id_pk PRIMARY KEY (id)
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE public.mdl_student_reports
    OWNER to esodbuser;

GRANT ALL ON TABLE public.mdl_student_reports TO esodbuser;

COMMENT ON TABLE public.mdl_student_reports
    IS 'Внешние отчёты студентов.';
